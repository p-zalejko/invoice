var params = {
  TableName: 'InvoiceApp_Users',
  Item: {
    'accountId' : {N: '1'},
    'username' : {S: 'foo'},
    'password' : {S: 'xx'},
    'roles' : {SS: ['USER']},
  },
};

// Call DynamoDB to add the item to the table
dynamodb.putItem(params, function(err, data) {
  if (err) {
    console.log("Error", err);
  } else {
    console.log("Success", data);
  }
});

// input for  yamitzky/dynamodb-gui
{
  "accountId": 2,
  "password": "abc",
  "roles": [
    "USER"
  ],
  "username": "abc"
}
